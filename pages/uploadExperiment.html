Move object commands

            function moveObj(obj, move) {

                // bấm lần nữa ngừng di chuyển
                obj.addEventListener("click", function () {
                    move = false
                })

                document.addEventListener("mousemove", function (event) {
                    // lấy pos chuột
                    let mX = event.clientX
                    let mY = event.clientY
            
                    if (move) {
                        obj.style.position = "absolute"
                        obj.style.margin = "0"

                        // để vị trí chuột ở trung tâm object nên chia 2
                        obj.style.top = mY - obj.getBoundingClientRect().height / 2 + "px"
                        obj.style.left = mX - obj.getBoundingClientRect().width / 2 + "px"
                    }
                })
            }

            //JS
            function bling(obj) {
                obj.style.animation = "bling 3s ease"
                obj.addEventListener("animationend", function () {
                    obj.style.animation = "none"
                })
            }

Alert function

            function tb(content, ms) {
                let tb = document.getElementById("alert");   //the id is changable
                tb.innerHTML = content;
                tb.style.display = "flex";

                setTimeout(function () {
                    tb.style.display = "none";
                }, ms)
            }
